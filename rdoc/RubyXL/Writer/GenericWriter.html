def render_xml
  seed_xml = Nokogiri::XML('<?xml version = "1.0" standalone ="yes"?>')
  seed_xml.encoding = 'UTF-8'

  yield(seed_xml)

  seed_xml.to_xml({ :indent => 0, :save_with => Nokogiri::XML::Node::SaveOptions::AS_XML })
end
